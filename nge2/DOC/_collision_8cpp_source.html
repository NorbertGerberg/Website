100eCollisionSide eCollision::GetCollisionSide(const eCollider::eAABB& box, const eCollider::eAABB& box2)
101{
102 real3 xOv = eTools::mini(box.mMax.x, box2.mMax.x) - eTools::maxi(box.mMin.x, box2.mMin.x);
103 real3 yOv = eTools::mini(box.mMax.y, box2.mMax.y) - eTools::maxi(box.mMin.y, box2.mMin.y);
104 real3 zOv = eTools::mini(box.mMax.z, box2.mMax.z) - eTools::maxi(box.mMin.z, box2.mMin.z);
105 real3 minOv = eTools::mini(eTools::mini(xOv, yOv), zOv);
106
107 const real3 epsilon = 0.0001f;
108
109 if (fabs(minOv - xOv) < epsilon)
110 return (box.mMin.x < box2.mMin.x) ? eCollisionSide::CsRight : eCollisionSide::CsLeft;
111 else if (fabs(minOv - yOv) < epsilon)
112 return (box.mMin.y < box2.mMin.y) ? eCollisionSide::CsTop : eCollisionSide::CsBottom;
113 else
114 return (box.mMin.z < box2.mMin.z) ? eCollisionSide::CsFront : eCollisionSide::CsBack;
115}

133const bool eCollision::OBBInOBB(const eCollider::eOBB& obb, const eCollider::eOBB& obb2)
134{
135 mat3 axes1 = mat3(obb.mAxes[0], obb.mAxes[1], obb.mAxes[2]);
136 mat3 axes2 = mat3(obb2.mAxes[0], obb2.mAxes[1], obb2.mAxes[2]);
137 mat3 axes = eMath::transpose(axes1) * axes2;
138
139 vec3 translation = obb2.mCenter - obb.mCenter;
140
141 for (int i = 0; i < 3; i++)
142 {
143 real3 t = eMath::dot(translation, axes[i]);
144
145 real3 obb1Proj = eMath::dot(glm::abs(obb.mAxes[i]), obb.mExtents);
146 real3 obb2Proj = eMath::dot(glm::abs(obb2.mAxes[i]), obb2.mExtents);
147
148 if (eMath::abs(t) > obb1Proj + obb2Proj)
149 return false;
150 }
151
152 axes = eMath::transpose(axes2) * axes1;
153
154 for (int i = 0; i < 3; i++)
155 {
156 real3 t = eMath::dot(translation, axes[i]);
157
158 real3 obb1Proj = eMath::dot(glm::abs(obb.mAxes[i]), obb.mExtents);
159 real3 obb2Proj = eMath::dot(glm::abs(obb2.mAxes[i]), obb2.mExtents);
160
161 if (eMath::abs(t) > obb1Proj + obb2Proj)
162 return false;
163 }
164
165 return true;
166}

169{
170 eCollider::eOBB obb;

56unsigned int parsePsid(unsigned char *buf, PsidHeader &psidHdr_)
57{
58 char *buffer = (char *) buf;
59
60 if (buf) {
61 memset(psidHdr_.title, 0, sizeof(psidHdr_.title));
62 strncpy(psidHdr_.title, buffer + PSID_NAME, 32);
63 psidHdr_.title[PSID_NAME + 32] = 0;
64 strncpy(psidHdr_.author, buffer + PSID_AUTHOR, 32);
65 psidHdr_.author[PSID_AUTHOR + 32] = 0;
66 strncpy(psidHdr_.copyright, buffer + PSID_COPYRIGHT, 32);
67 psidHdr_.copyright[PSID_COPYRIGHT + 32] = 0;
68 } else {
69 memset(psidHdr_.title, 0, sizeof(psidHdr_.title));
70 memset(psidHdr_.author, 0, 32);

126unsigned int readFile(char *fName, unsigned char **bufferPtr, size_t *bLen)

130 if (!fName)
131 return 2;

141 std::fseek(file, 0, SEEK_END);
142 flen = std::ftell(file);

151 std::cerr << " exception: " << str << std::endl;
152 return 1;
153 }

157bool psidChangeTrack(int direction)
158{
159 if (direction > 0) {
160 if (psidHdr.tracks > psidHdr.current) {

162 rsidPlayer[1] += direction;
163 psidHdr.current += direction;

165 std::cerr << "No more tracks." << std::endl;
166 return false;

168 } else {

171 rsidPlayer[1] += direction;
172 psidHdr.current += direction;
173 } else {
174 std::cerr << "No more tracks." << std::endl;
175 return false;
176 }
177 }
178 if (psidHdr.type == 1 || (psidHdr.type == 2 && !psidHdr.replayAddress)) {
179 ted->writeProtectedPlayerMemory(playerStartAddress, rsidPlayer, sizeof(rsidPlayer));
180 } else {
181 ted->writeProtectedPlayerMemory(playerStartAddress, psidPlayer, sizeof(psidPlayer));
182 }
183 cpu->setPC(playerStartAddress);
184 return true;

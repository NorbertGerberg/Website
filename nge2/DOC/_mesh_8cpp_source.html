43void eMesh::Load(eString fileName)
44{
45 Assimp::Importer importer;
46 uint flags = aiProcess_FlipUVs
47 | aiProcess_Triangulate
48 | aiProcess_GenSmoothNormals
49 | aiProcess_CalcTangentSpace
50 | aiProcess_SplitLargeMeshes
51 | aiProcess_OptimizeMeshes;
52
53 const aiScene* scene = importer.ReadFile(fileName.Get(), flags);
54
55 if (!scene || scene->mFlags & AI_SCENE_FLAGS_INCOMPLETE || !scene->mRootNode)
56 {
57 eConsole::PrintLog("eMesh::Load", "Failed to load model at '" + fileName.GetStrg() + "'!");
58 eConsole::WriteToDisk();
59 return;
60 }
61
62 ProcessNode(scene->mRootNode, scene);
63}

102eMeshContainer eMesh::ProcessMesh(aiMesh* mesh, const aiScene* scene)
103{
104 std::vector<eMeshVertex> vertices;
105 std::vector<uint16> indices;
106
107 for (uint i = 0; i < mesh->mNumVertices; i++)
108 {
109 eMeshVertex vertex;
110
111 vertex.x = mesh->mVertices[i].x;
112 vertex.y = mesh->mVertices[i].y;
113 vertex.z = mesh->mVertices[i].z;
114
115 if (mesh->HasNormals())
116 {
117 vertex.nx = mesh->mNormals[i].x;
118 vertex.ny = mesh->mNormals[i].y;
119 vertex.nz = mesh->mNormals[i].z;
120 }
121
122 if (mesh->mTextureCoords[0])
123 {
124 vertex.u = mesh->mTextureCoords[0][i].x;
125 vertex.v = mesh->mTextureCoords[0][i].y;
126
127 vertex.tx = mesh->mTangents[i].x;
128 vertex.ty = mesh->mTangents[i].y;
129 vertex.tz = mesh->mTangents[i].z;
130
131 vertex.bx = mesh->mBitangents[i].x;
132 vertex.by = mesh->mBitangents[i].y;
133 vertex.bz = mesh->mBitangents[i].z;
134 }
135 else
136 {
137 vertex.u = 0.0f;
138 vertex.v = 0.0f;
139 }
140 vertices.push_back(vertex);
141 }
142
143 for (uint i = 0; i < mesh->mNumFaces; i++)
144 {
145 aiFace face = mesh->mFaces[i];
146 for (uint j = 0; j < face.mNumIndices; j++)
147 indices.push_back(face.mIndices[j]);
148 }
149
150 return CreateMesh(vertices, indices);
151}

154eMeshContainer eMesh::CreateMesh(std::vector<eMeshVertex> vertices, std::vector<uint16> indices)
155{
156 //Meshoptimizer code
157 std::vector<eMeshVertex> _rt_vertices = vertices;
158 std::vector<uint16> _rt_indices = indices;
159
160 if (bOptimize)
161 {
162 meshopt_optimizeVertexCache(&_rt_indices[0], &_rt_indices[0], _rt_indices.size(), _rt_vertices.size());
163 meshopt_optimizeVertexFetch(&_rt_vertices[0], &_rt_indices[0], _rt_indices.size(), &_rt_vertices[0], _rt_vertices.size(), sizeof(eMeshVertex));
164 }
165
166 eMeshContainer rtVl;
167 rtVl.mVertices = _rt_vertices;
168 rtVl.mIndices = _rt_indices;
169
170 eVertexLayout vlayout;
171 vlayout.begin()
172 .add(bgfx::Attrib::Position, 3, bgfx::AttribType::Float)
173 .add(bgfx::Attrib::Normal, 3, bgfx::AttribType::Float)
174 .add(bgfx::Attrib::TexCoord0, 2, bgfx::AttribType::Float)
175 .add(bgfx::Attrib::Tangent, 3, bgfx::AttribType::Float)
176 .add(bgfx::Attrib::Bitangent, 3, bgfx::AttribType::Float)
177 .end();
178
179 rtVl.mVbh = eShader::CreateVertexBuffer(&vertices[0], vertices.size() * sizeof(eMeshVertex), vlayout);
180 if (!bgfx::isValid(rtVl.mVbh))
181 {
182 eConsole::PrintLog("eMesh::CreateMesh", "Failed to create vbh.");
183 eConsole::WriteToDisk();
184 return {};
185 }
186
187 rtVl.mIbh = eShader::CreateIndexBuffer(&indices[0], indices.size() * sizeof(uint16));
188 if (!bgfx::isValid(rtVl.mIbh))
189 {
190 eConsole::PrintLog("eMesh::CreateMesh", "Failed to create ibh.");
191 eConsole::WriteToDisk();
192 return {};
193 }
194 return rtVl;
195}

334const int eMesh::DrawInsBegin(std::vector<eTransformation3D>& transformations, bgfx::InstanceDataBuffer& idb)
335{
336 const int insCnt = transformations.size();
337 const uint16 insStride = 64;
338 uint32 drawnIns = bgfx::getAvailInstanceDataBuffer(insCnt, insStride);
339
340 const int couldNotDraw = insCnt - drawnIns;
341
342 bgfx::allocInstanceDataBuffer(&idb, drawnIns, insStride);
343
344 uint8* data = idb.data;
345
346 for (uint32 i = 0; i < drawnIns; i++)
347 {
348 const mat4 mtx = eTools::Transform(transformations[i]);
349 const float* _mtx = eMath::value_ptr(mtx);
350 memcpy(data, _mtx, insStride);
351 data += insStride;
352 }
353
354 return couldNotDraw;
355}

358void eMesh::DrawInsEnd(eMeshContainer& container, eTransformation3D& first, eLayer* layer, eMaterial* mat, bgfx::InstanceDataBuffer& idb)
359{
360 eShader* shader = mDrawFnc(container, layer, first, mat);
361
362 shader->ApplyVertexBuffer(0, container.mVbh);
363 shader->ApplyIndexBuffer(container.mIbh);
364
365 bgfx::setInstanceDataBuffer(&idb);
366
367 eMaterialPrp* matPrp = mat->GetProperties();
368 if (matPrp->bTwoSides)
369 {
370 eGraphics::SetState(BGFX_STATE_BLEND_ALPHA
371 | BGFX_STATE_WRITE_RGB
372 | BGFX_STATE_WRITE_A
373 | BGFX_STATE_WRITE_Z
374 | BGFX_STATE_DEPTH_TEST_LESS
375 | BGFX_STATE_MSAA);
376 }
377 else
378 eGraphics::SetState();
379
380 shader->Submit(layer->GetProperties().mViewId, false);
381}

390bool eMesh::DrawSingleBegin(eMeshContainer& container, eLayer* layer, eTransformation3D transformation, eMaterial* mat)
391{
392 eViewport3D* viewport = layer->GetViewport();
393 if (viewport == nullptr)
394 return false;
395
396 eMaterial* useMat = container.mMat;
397
398 bool noMat = true;
399 if (mat != nullptr)
400 {
401 noMat = false;
402 useMat = mat;
403 }
404 else if (container.mMat != nullptr)
405 noMat = false;
406
407 eMaterialPrp* matPrp = useMat->GetProperties();
408
409 if (matPrp->bTwoSides)
410 {
411 eGraphics::SetState(BGFX_STATE_BLEND_ALPHA
412 | BGFX_STATE_WRITE_RGB
413 | BGFX_STATE_WRITE_A
414 | BGFX_STATE_WRITE_Z
415 | BGFX_STATE_DEPTH_TEST_LESS
416 | BGFX_STATE_MSAA);
417 }
418 else
419 eGraphics::SetState();
420
421 mat4 mdl = eTools::Transform(transformation);
422 bgfx::setTransform(eMath::value_ptr(mdl));
423
424 return !noMat;
425}

66 BackendData()

71 ~BackendData()
72 {
73 // Wait until thread is done.
74 threadrun++;
75 while (threadrun == 1)
76 {
77 Thread::sleep(10);
78 }
79
80 if(playerObj)
81 {
82 (*playerObj)->Destroy(playerObj);
83 }
84
85 if(outputMixObj)
86 {
87 (*outputMixObj)->Destroy(outputMixObj);
88 }
89
90 if(engineObj)
91 {
92 (*engineObj)->Destroy(engineObj);
93 }
94
95 for(int idx = 0; idx < NUM_BUFFERS; ++idx)
96 {
97 delete [] outputBuffers[idx];
98 }
99 }

136 static void opensles_iterate(SoLoud::Soloud *aSoloud)
137 {
138 BackendData *data = static_cast<BackendData*>(aSoloud->mBackendData);
139
140 // If we have no buffered queued, queue one up for playback.
141 if(data->buffersQueued == 0)
142 {
143 // Get next output buffer, advance, next buffer.
144 short * outputBuffer = data->outputBuffers[data->activeBuffer];
145 data->activeBuffer = (data->activeBuffer + 1) % NUM_BUFFERS;
146 short * nextBuffer = data->outputBuffers[data->activeBuffer];
147
148 // Mix this buffer.
149 const int bufferSizeBytes = data->bufferSize * data->channels * sizeof(short);
150 (*data->playerBufferQueue)->Enqueue(data->playerBufferQueue, outputBuffer, bufferSizeBytes);
151 ++data->buffersQueued;
152
153 aSoloud->mixSigned16(nextBuffer,data->bufferSize);
154 }
155 }

157 static void opensles_thread(void *aParam)
158 {
159 Soloud *soloud = static_cast<Soloud*>(aParam);
160 BackendData *data = static_cast<BackendData*>(soloud->mBackendData);
161 while (data->threadrun == 0)
162 {
163 opensles_iterate(soloud);
164
165 // TODO: Condition var?
166 Thread::sleep(1);
167 }
168 data->threadrun++;
169 }

181 result opensles_init(SoLoud::Soloud *aSoloud, unsigned int aFlags, unsigned int aSamplerate, unsigned int aBuffer, unsigned int aChannels)
182 {
183 BackendData *data = new BackendData();
184
185 // Allocate output buffer to mix into.
186 data->bufferSize = aBuffer;
187 data->channels = aChannels;
188 const int bufferSizeBytes = data->bufferSize * data->channels * sizeof(short);
189 for(int idx = 0; idx < NUM_BUFFERS; ++idx)
190 {
191 data->outputBuffers[idx] = new short[ data->bufferSize * data->channels ];
192 memset(data->outputBuffers[idx], 0, bufferSizeBytes);
193 }
194
195 // Create engine.
196 const SLEngineOption opts[] = { SL_ENGINEOPTION_THREADSAFE, SL_BOOLEAN_TRUE };
197 if(slCreateEngine( &data->engineObj, 1, opts, 0, NULL, NULL ) != SL_RESULT_SUCCESS)
198 {
199 LOG_ERROR( "Failed to create engine." );
200 return UNKNOWN_ERROR;
201 }
202
203 // Realize and get engine interfaxce.
204 (*data->engineObj)->Realize(data->engineObj, SL_BOOLEAN_FALSE);
205 if((*data->engineObj)->GetInterface(data->engineObj, SL_IID_ENGINE, &data->engine) != SL_RESULT_SUCCESS)
206 {
207 LOG_ERROR( "Failed to get engine interface." );
208 return UNKNOWN_ERROR;
209 }
210
211 // Create output mix.
212 const SLInterfaceID ids[] = { SL_IID_VOLUME };
213 const SLboolean req[] = { SL_BOOLEAN_FALSE };
214
215 if((*data->engine)->CreateOutputMix(data->engine, &data->outputMixObj, 1, ids, req) != SL_RESULT_SUCCESS)
216 {
217 LOG_ERROR( "Failed to create output mix object." );
218 return UNKNOWN_ERROR;
219 }
220 (*data->outputMixObj)->Realize(data->outputMixObj, SL_BOOLEAN_FALSE);
221
222 if((*data->outputMixObj)->GetInterface(data->outputMixObj, SL_IID_VOLUME, &data->outputMixVol) != SL_RESULT_SUCCESS)
223 {
224 LOG_INFO( "Failed to get output mix volume interface." );
225 }
226
227 // Create android buffer queue.
228 data->inLocator.locatorType = SL_DATALOCATOR_ANDROIDSIMPLEBUFFERQUEUE;
229 data->inLocator.numBuffers = NUM_BUFFERS;
230
231 // Setup data format.
232 SLDataFormat_PCM format;
233 format.formatType = SL_DATAFORMAT_PCM;
234 format.numChannels = data->channels;
235 format.samplesPerSec = aSamplerate * 1000; //mHz
236 format.bitsPerSample = SL_PCMSAMPLEFORMAT_FIXED_16;
237 format.containerSize = 16;
238 format.endianness = SL_BYTEORDER_LITTLEENDIAN;
239
240 // Determine channel mask.
241 if(data->channels == 2)
242 {
243 format.channelMask = SL_SPEAKER_FRONT_LEFT | SL_SPEAKER_FRONT_RIGHT;
244 }
245 else
246 {
247 format.channelMask = SL_SPEAKER_FRONT_CENTER;
248 }
249
250 SLDataSource src;
251 src.pLocator = &data->inLocator;
252 src.pFormat = &format;
253
254 // Output mix.
255 data->outLocator.locatorType = SL_DATALOCATOR_OUTPUTMIX;
256 data->outLocator.outputMix = data->outputMixObj;
257
258 data->dstDataSink.pLocator = &data->outLocator;
259 data->dstDataSink.pFormat = NULL;
260
261 // Setup player.
262 {
263 const SLInterfaceID ids[] = { SL_IID_VOLUME, SL_IID_ANDROIDSIMPLEBUFFERQUEUE };
264 const SLboolean req[] = { SL_BOOLEAN_TRUE, SL_BOOLEAN_TRUE };
265
266 (*data->engine)->CreateAudioPlayer(data->engine, &data->playerObj, &src, &data->dstDataSink, sizeof(ids) / sizeof(ids[0]), ids, req);
267
268 (*data->playerObj)->Realize(data->playerObj, SL_BOOLEAN_FALSE);
269
270 (*data->playerObj)->GetInterface(data->playerObj, SL_IID_PLAY, &data->player);
271 (*data->playerObj)->GetInterface(data->playerObj, SL_IID_VOLUME, &data->playerVol);
272
273 (*data->playerObj)->GetInterface(data->playerObj, SL_IID_ANDROIDSIMPLEBUFFERQUEUE, &data->playerBufferQueue);
274 }
275
276 aSoloud->mBackendData = data; // Must be set before callback
277
278 // Begin playing.
279 {
280 const int bufferSizeBytes = data->bufferSize * data->channels * sizeof(short);
281 (*data->playerBufferQueue)->Enqueue(data->playerBufferQueue, data->outputBuffers[0], bufferSizeBytes);
282 data->activeBuffer = (data->activeBuffer + 1) % NUM_BUFFERS;
283
284 (*data->player)->RegisterCallback(data->player, soloud_opensles_play_callback, aSoloud);
285 (*data->player)->SetCallbackEventsMask(data->player, SL_PLAYEVENT_HEADATEND);
286 (*data->player)->SetPlayState(data->player, SL_PLAYSTATE_PLAYING);
287
288 }
289
290 //
291 aSoloud->postinit_internal(aSamplerate,data->bufferSize,aFlags,2);
292 aSoloud->mBackendCleanupFunc = soloud_opensles_deinit;
293
294 LOG_INFO( "Creating audio thread." );
295 Thread::createThread(opensles_thread, (void*)aSoloud);
296
297 aSoloud->mBackendString = "OpenSL ES";
298 return SO_NO_ERROR;
299 }

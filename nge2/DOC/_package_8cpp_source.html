57void ePackage::Load()
58{
59 EDR::Archive* archive = EDR::OpenArchive(mTotPath.Get());
60
61 if (archive == nullptr)
62 {
63 eConsole::PrintLog("WARNING::ePackage::Load", "Package '" + mName.GetStrg() + "' could not be found!");
64 eConsole::WriteToDisk();
65 return;
66 }
67
68 strg info = EDR::CopyArchiveFileContents(archive, "info.rd");
69 int amount = EDR::GroupItemAmountI(info, mName.GetStrg());
70
71 for (int i = 0; i < amount; i++)
72 {
73 strg group = EDR::GroupItemNameI(info, mName.GetStrg(), i);
74 LoadAsset(archive, group);
75 }
76 archive = EDR::CloseArchive(archive);
77}

87void ePackage::ReleaseAsset(eString name)
88{
89 bool isValid = false;
90 for (uint i = 0; i < mAssets.Size(); i++)
91 {
92 auto& asset = mAssets[i];
93 eString& assName = mAssets.IdAt(i);
94
95 if (!assName.Compare(name))
96 continue;
97
98 for (auto& it : mFncs)
99 {
100 if (asset.mType.Compare(it.first))
101 {
102 it.second.mDeconstr(asset.mValue);
103 asset.mType.Clear();
104 isValid = true;
105 break;
106 }
107 }
108 if (isValid) break;
109 }
110
111 if (isValid)
112 mAssets.Pop(name);
113}

115void ePackage::InsertAsset(eString originalFilePath, eString destName, eString infoFileContent, eString assetType)
116{

129
130 strg out_inf = "<" + mName.GetStrg() + ">\n";
131 for(auto& it : mAssets.Geta())
132 out_inf += "<" + it.first.GetStrg() + ">\n";
133 out_inf += "<" + destName.GetStrg() + ">\n";
134 out_inf += "</" + mName.GetStrg() + ">\n";
135 EDR::RemoveFileFromArchive(archive, "info.rd");
136 EDR::AddFileToArchive(archive, "info.rd", out_inf.c_str());

173void ePackage::LoadAsset(EDR::Archive* archive, strg name)
174{
175 if (mAssets.Find(name)) return;
176
177 strg infoFile = name + ".rd";
178 strg infoText = EDR::CopyArchiveFileContents(archive, infoFile.c_str());
179 eString assetType = EDR::GetStringI(infoText, "type");
180 assetType.Replace('|', ' ');
181
182 for (auto& it : mFncs)
183 {
184 if (assetType.Compare(it.first))
185 {
186 eAsset rtVl;
187 rtVl.mType = assetType;
188 rtVl.mValue = it.second.mConstr(infoText, name, archive);
189 mAssets.Push(rtVl, name);
190 break;
191 }
192 }
193}

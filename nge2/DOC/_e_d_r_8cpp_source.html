260void EDR::ConvertToBinary(strg& text)
261{
262 strg rtBin = "";
263 for (uint i = 0; i < text.length(); i++)
264 {
265 //Convert character to ASCII code
266 int ascii = (int)text[i];
267 strg _tmpBin = "";
268 while (ascii > 0)
269 {
270 if (ascii % 2 == 0)
271 _tmpBin = "0" + _tmpBin;
272 else
273 _tmpBin = "1" + _tmpBin;
274 ascii /= 2;
275 }
276
277 //Pad with zeros to ensure each binary code is 8 bits
278 while (_tmpBin.length() < 8)
279 _tmpBin = "0" + _tmpBin;
280 rtBin += _tmpBin;
281 }
282
283 //Compress
284 strg crtBin;
285 uint zeros = 0;
286 uint ones = 0;
287 bool isCounting0 = false;
288 bool firstCount = false;
289 for (auto& it : rtBin)
290 {
291 if (it == '0')
292 {
293 if (!isCounting0 && firstCount)
294 {
295 isCounting0 = true;
296 crtBin += std::to_string(ones);
297 ones = 0;
298 }
299
300 if (!firstCount)
301 {
302 firstCount = true;
303 crtBin = "0";
304 }
305 zeros += 1;
306 }
307 else if (it == '1')
308 {
309 if (isCounting0 && firstCount)
310 {
311 isCounting0 = false;
312 crtBin += std::to_string(zeros);
313 zeros = 0;
314 }
315
316 if (!firstCount)
317 {
318 firstCount = true;
319 crtBin = "1";
320 }
321 ones += 1;
322 }
323 }
324 text = crtBin;
325 return;
326}

329void EDR::ConvertToText(strg& binary)
330{
331 //De-Compress
332 strg ucbinary;
333 bool isCounting0 = false;
334 if (binary.at(0) == '0')
335 isCounting0 = true;
336
337 for (auto& it : binary)
338 {
339 uint amount = ::atoi(strg(1, it).c_str());
340 strg out = isCounting0 ? "0" : "1";
341 isCounting0 = !isCounting0;
342
343 for (uint i = 0; i < amount; i++)
344 ucbinary += out;
345 }
346
347 strg rtTxt = "";
348 for (uint i = 0; i < ucbinary.length(); i += 8)
349 {
350 strg byteString = ucbinary.substr(i, 8);
351 int decimal = 0;
352 for (int f = 0; f < byteString.length(); f++)
353 decimal = decimal * 2 + (byteString[f] - '0');
354 char character = (char)decimal;
355 rtTxt += character;
356 }
357 binary = rtTxt;
358 return;
359}
